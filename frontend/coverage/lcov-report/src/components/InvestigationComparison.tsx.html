// Comparison Row Component
function ComparisonRow({
  label,
  values,
  type = 'text',
  highlight,
  format
}: {
  label: string;
  values: any[];
  type?: 'text' | 'number' | 'status';
  highlight?: 'max' | 'min';
  format?: (value: any) => string;
}) {
  const maxValue = type === 'number' ? Math.max(...values) : null;
  const minValue = type === 'number' ? Math.min(...values) : null;
 
  return (
    <tr>
      <td className="px-6 py-4 whitespace-nowrap text-sm font-medium text-gray-900 dark:text-white">
        {label}
      </td>
      {values.map((value, index) => {
        const isHighlighted =
          (highlight === 'max' && value === maxValue) ||
          (highlight === 'min' && value === minValue);
 
        return (
          <td
            key={index}
            className={`px-6 py-4 whitespace-nowrap text-sm ${
              isHighlighted
                ? 'bg-green-50 dark:bg-green-900/20 font-bold text-green-600'
                : 'text-gray-700 dark:text-gray-300'
            }`}
          >
            {format ? format(value) : value}
            {isHighlighted && ' üèÜ'}
          </td>
        );
      })}
    </tr>
  );
}
